# TabbedChips

Tab chips are to be used in a filter setting where the user is given the ability to refine the data source they are accessing.

## Import

```tsx
import { TabbedChips } from '@coinbase/cds-mobile/chips/TabbedChips'
```

## Examples

### Basic Example

```jsx
function TabbedChipsExample() {
  const tabs = [
    {
      id: 'all_tab',
      label: 'All',
    },
    {
      id: 'swap_',
      label: 'Swap',
    },
    {
      id: 'collect_tab',
      label: 'Collect',
    },
    {
      id: 'bridge_tab',
      label: 'Bridge',
    },
  ];
  const [value, setValue] = useState(tabs[0].id);
  return (
    <VStack maxWidth={375} borderRadius={400} bordered paddingX={2} gap={2} paddingY={2}>
      <VStack paddingTop={1}>
        <Text as="p" font="title3">
          Popular assets
        </Text>
      </VStack>
      <SearchInput compact placeholder="Search by name" />
      <TabbedChips value={value} onChange={setValue} tabs={tabs} />
    </VStack>
  );
}
```
